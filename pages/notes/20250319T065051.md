```emacs-lisp
;; defsubst
(defun gptel--perplexity-parse-citations (citations)
  (let ((counter 0))
    (concat "\n\n- Citations:\n"
            (mapconcat (lambda (url)
                         (setq counter (1+ counter))
                         (format "  [%d] %s" counter (decode-coding-string (url-unhex-string url t) 'utf-8)))
                       citations "\n"))))
```
