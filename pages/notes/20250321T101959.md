;; https://openrouter.ai/
(gptel-make-openai "OpenRouter"
  :host "openrouter.ai"
  :endpoint "/api/v1/chat/completions"
  :stream t
  :key #'gptel-api-key
  :request-params '(:temperature 0.0)
  :models '(
            google/gemini-2.5-flash
